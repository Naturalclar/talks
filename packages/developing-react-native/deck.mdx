```jsx subtitle="Regular React"
import React from "react";

type Props = { name: string };

export const HelloWorld = ({ name }: Props) => (
  <div>
    <p>Hello {name}!</p>
  </div>
);
```

```jsx subtitle="In React Native"
import React from "react";
import { View, Text } from "react-native";

type Props = { name: string };

export const HelloWorld = ({ name }: Props) => (
  <View>
    <Text>Hello {name}!</Text>
  </View>
);
```
